D.2 Символьные производные

Символьные вычисления вместе с гессианом и градиентом

stats::deriv(~ -(y * (b0 + x * b1) - log(1 + exp(b0 + b1 * x))),
             c("b0", "b1"))
expression({
    .expr6 <- exp(b0 + b1 * x)
    .expr7 <- 1 + .expr6
    .value <- -(y * (b0 + x * b1) - log(.expr7))
    .grad <- array(0, c(length(.value), 2L), list(NULL, c("b0", 
        "b1")))
    .grad[, "b0"] <- -(y - .expr6/.expr7)
    .grad[, "b1"] <- -(y * x - .expr6 * x/.expr7)
    attr(.value, "gradient") <- .grad
    .value
})
